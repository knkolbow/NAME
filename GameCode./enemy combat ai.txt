public abstract class EnemyCombatStrategy{
  \\This is the class that the character passes through, this is an
  \\abstract class mainly used to get the character class into the
  \\other strategy classes
 
  public abstract void TakeTurn(PlayerCharacter player, EnemyCharacter enemy);
}

public class PlayerAggressiveStrategy : CombatStrategy{
  \\the aggressive strategy inherits CombatStrategy
  \\this strategy picks the highest damage attack and does it

  public override void TakeTurn(PlayerCharacter player, EnemyCharacter enemy){
	if (player.health > 0){
		if (enemy.mana >= 10){
			enemy.Ult()
		}
		else {
			player.Attack()
		}
	}
